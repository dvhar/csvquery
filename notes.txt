building on windows using msys2:

	pacman -S mingw-w64-x86_64-cmake mingw-w64-x86_64-intel-tbb mingw-w64-x86_64-boost mingw-w64-x86_64-toolchain mingw-w64-x86_64-python-pip
	python3 -m pip install pyyaml

	git clone https://github.com/dvhar/csvquery
	cd csvquery/build
	cmake .. -G "MinGW Makefiles" -DCMAKE_SH="CMAKE_SH-NOTFOUND"
	mingw32-make.exe -j4

embed website:
	./deps/embed.sh
